N_SAMPLES = 100

CLASS_0 = ( (2, 3), (0.8, 2.5))
CLASS_1 = ( (5, 6), (1.2, 1.9))
CLASS_2 = ( (8, 1), (0.9, 0.9))
CLASS_3 = ((15, 4), (0.5, 2.0))

def generate_data(CLASS_data, n_samples) -> np.ndarray[np.ndarray, np.ndarray]:
    dim_0 = np.random.normal(loc=CLASS_data[0][0], scale=CLASS_data[1][0], size=n_samples)
    dim_1 = np.random.normal(loc=CLASS_data[0][1], scale=CLASS_data[1][1], size=n_samples)
    return np.array(list(zip(dim_0, dim_1)))

class_0_data = generate_data(CLASS_0, N_SAMPLES)
class_1_data = generate_data(CLASS_1, N_SAMPLES)
class_2_data = generate_data(CLASS_2, N_SAMPLES)
class_3_data = generate_data(CLASS_3, N_SAMPLES)

class_A_mean = np.zeros(5)
class_B_mean = np.array([1.5] * 5)

class_A_cov = np.array([
    [1,0.8,0.1,0,0],
    [0.8,1,0.3,0,0],
    [0.1,0.3,1,0.5,0],
    [0,0,0.5,1,0.2],
    [0,0,0,0.2,1]
])

class_B_cov = np.array([
    [1,-0.7,0.2,0,0],
    [-0.7,1.5,0.4,0,0],
    [0.2,0.4,1.5,0.6,0],
    [0,0,0.6,1.5,0.3],
    [0,0,0,0.3,1.5]
])

class_A_data = np.random.multivariate_normal(mean=class_A_mean, cov=class_A_cov, size=500)
class_B_data = np.random.multivariate_normal(mean=class_B_mean, cov=class_B_cov, size=500)

pca = PCA(n_components=2)
class_A_data_2d = pca.fit_transform(class_A_data)
class_B_data_2d = pca.transform(class_B_data)

plt.scatter(class_A_data_2d[:,0], class_A_data_2d[:,1], color='red', label='Classe A', alpha=0.1)
plt.scatter(class_B_data_2d[:,0], class_B_data_2d[:,1], color='blue', label='Classe B', alpha=0.1)
plt.xlabel('Componente Principal 1')
plt.ylabel('Componente Principal 2')
plt.title('Dados Projetados em 2D via PCA')
plt.legend()
plt.show()

    df_train.loc[df_train[column].isnull(), column] = np.random.choice(
        proportions.index, size=missing_count, p=proportions.values
    )
